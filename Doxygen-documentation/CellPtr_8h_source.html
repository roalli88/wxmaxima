375  inline Observed *base_get() const noexcept
376  {
377  // Warning: This function is CRITICAL to the performance of wxMaxima
378  // as a whole!
379  //
380  // The common hot path that iterates cells via the m_nextToDraw uses
381  // this function and is intimately tied to its performance. Small
382  // changes here can cause performance regressions - or small performance
383  // improvements.
384  //
385  // If you change anything, do before- and after- measurements to verify
386  // that whatever improvement you sought is in fact achieved. Changes that
387  // don't measurably improve performance are discouraged.
388 
389  // The common path, meant to be hot: if we point directly at the observed object,
390  // just return that. This is also where null is returned if the pointer is null.
391  // `HasObserved()` is a simple bitmask test, and `CastAsObserved` is a binary
392  // NO-OP.
393 
394  if (m_ptr.HasObserved())
395  return m_ptr.CastAsObserved();
396 
397  // Otherwise, we must be pointing to a control block: get the pointed-to
398  // observed from the control block. Since such use case is meant to be
399  // rare, the overhead of pointer chasing (one extra layer of indirection)
400  // is acceptable.
401 
402  auto *const observed = m_ptr.CastAsControlBlock()->Get();
403  if (observed)
404  return observed;
405 
406  // We have a control block, but the observed object is gone: we dereference
407  // the zombie control block, to deallocate it as soon as possible, and we
408  // reset ourselves to null. This happens only once per observed object, and
409  // subsequent calls will go via the common path.
410 
411  return DerefControlBlock(); // returns null - allows a tail call optimization
412  }
